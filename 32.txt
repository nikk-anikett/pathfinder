function createSpaceBetweenRows(data) {
  const newData = [];
  for (let i = 0; i < data.length - 1; i++) {
    newData.push(data[i]);
    if (data[i].ClientName !== data[i + 1].ClientName) {
      newData.push({
        ClientName: "",

        Exec_Exp_Net: 0,
        GroupCode: null,
        ExpLimit_In_Cr: "",
